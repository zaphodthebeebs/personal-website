# Add this to your existing Caddyfile for mabus.ai

mabus.ai, www.mabus.ai {
    # API proxy - must come BEFORE the main site proxy
    handle /api/* {
        reverse_proxy localhost:3301
    }

    # Main website
    reverse_proxy localhost:3300
}
